


.currentPiece: rb 4*4

.tempPiece: rb 4*4


; Inputs:
;	HL: source addr (line of 4 blocks)
;	DE: destiny addr (column of 4 blocks)
RotatePiece:

ld	b, 4
.loop:
	ld	a, (hl)
	ld	(de), a

	; de = de + 4
	inc	de
	inc	de
	inc	de
	inc	de

	inc	hl

	djnz	.loop

ret
